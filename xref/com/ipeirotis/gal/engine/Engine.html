117 	public void execute() {
118 		setCategories(loadCategories(ctx.getCategoriesFile()));
119 
120 		setDawidSkene(new DawidSkene(getCategories()));
121 		
122 		if (getDs().fixedPriors() == true)
123 			println("Using fixed priors.");
124 		else
125 			println("Using data-inferred priors.");
126 
127 		if (ctx.hasCosts()) {
128 			setCosts(loadCosts(ctx.getCostFile()));
129 
130 			for (MisclassificationCost mcc : getCosts()) {
131 				getDs().addMisclassificationCost(mcc);
132 			}
133 		}
134 
135 		setLabels(loadWorkerAssignedLabels(ctx.getInputFile()));
136 
137 		int al = 0;
138 
139 		for (AssignedLabel l : getLabels()) {
140 			if (++al % 1000 == 0)
141 				print(".");
142 			getDs().addAssignedLabel(l);
143 		}
144 		println("%d worker-assigned labels loaded.", getLabels().size());
145 
146 		if (ctx.hasGoldFile()) {
147 			setCorrect(loadGoldLabels(ctx.getGoldFile()));
148 
149 			int cl = 0;
150 			for (CorrectLabel l : getCorrect()) {
151 				if (++cl % 1000 == 0)
152 					print(".");
153 				getDs().addCorrectLabel(l);
154 			}
155 			println("%d correct labels loaded.", getCorrect().size());
156 		}
157 
158 		if (ctx.hasEvaluations()) {
159 			setEvaluation(loadEvaluationLabels(ctx.getEvaluationFile()));
160 			int el = 0;
161 			for (CorrectLabel l : getEvaluation()) {
162 				if (++el % 1000 == 0)
163 					print(".");
164 				getDs().addEvaluationLabel(l);
165 			}
166 			println(getEvaluation().size() + " evaluation labels loaded.");
167 		}
168 
169 		// We compute the evaluation-based confusion matrix for the workers
170 		getDs().evaluateWorkers();
171 
172 		println("");
173 		println("Running the Dawid&Skene algorithm");
174 		double epsilon = ctx.getEpsilon();
175 		int maxIterations = ctx.getNumIterations();
176 		double ll = getDs().getLogLikelihood();
177 		println("Initial Log-likelihood: %3.6f", ll);
178 		ll = getDs().estimate(maxIterations, epsilon);
179 		println("Final Log-likelihood: %3.6f", ll);
180 		println("Done\n");
181 
182 		if (ctx.hasEvaluateResultsAgainstFile()) {
183 			rptCtx.setExpectedEvaluation(loadGoldLabels(ctx
184 					.getEvaluateResultsAgainstFile()));
185 		}
186 
187 		executeReports();
188 	}

190 	private void executeReports() {
191 		if (!ctx.isDryRun()) {
192 			reports.add(new CategoryPriorsReport());
193 		}
194 		
195 		reports.add(new SummaryReport());
196 		reports.add(new ConfusionMatrixReport());
197 		
198 		try {
199 			File outputDir = new File("results");
200 			
201 			if (! outputDir.exists())
202 				outputDir.mkdir();
203 			
204 			
205 			for (Report report : reports) {
206 				report.execute(rptCtx);
207 			}
208 		} catch (IOException exc) {
209 			throw new RuntimeException(exc);
210 		}
211 	}

242 	public Set<AssignedLabel> loadAssignedLabels(String[] lines) {
243 
244 		Set<AssignedLabel> labels = new HashSet<AssignedLabel>();
245 		int cnt = 1;
246 		for (String line : lines) {
247 			String[] entries = line.split("\t");
248 			if (entries.length != 3) {
249 				throw new IllegalArgumentException(
250 						"Error while loading from assigned labels file (line #"
251 								+ cnt + "): " + line);
252 			}
253 			cnt++;
254 
255 			String workername = entries[0];
256 			String objectname = entries[1];
257 			String categoryname = entries[2];
258 
259 			AssignedLabel al = new AssignedLabel(workername, objectname,
260 					categoryname);
261 			labels.add(al);
262 		}
263 		return labels;
264 	}

266 	public Set<Category> loadCategories(String[] lines) {
267 
268 		Set<Category> categories = new HashSet<Category>();
269 		for (String line : lines) {
270 			// First we check if we have fixed priors or not
271 			// If we have fixed priors, we have a TAB character
272 			// after the name of each category, followed by the prior value
273 			String[] l = line.split("\t");
274 			if (l.length == 1) {
275 				Category c = new Category(line);
276 				categories.add(c);
277 			} else if (l.length == 2) {
278 				String name = l[0];
279 				Double prior = new Double(l[1]);
280 				Category c = new Category(name);
281 				c.setPrior(prior);
282 				categories.add(c);
283 			}
284 		}
285 		return categories;
286 	}
