27  	public static String cleanLine(String line) {
28  
29  		StringBuffer buffer = new StringBuffer();
30  		for (int i = 0; i < line.length(); i++) {
31  			char c = line.charAt(i);
32  			if (c < 128 && (Character.isLetter(c) || Character.isDigit(c))) {
33  				buffer.append(c);
34  			} else {
35  				// buffer.append('');
36  			}
37  		}
38  		return buffer.toString().toLowerCase();
39  	}

41  	public static String getFile(String FileName) {
42  
43  		StringBuffer buffer = new StringBuffer();
44  
45  		try {
46  			BufferedReader dataInput = new BufferedReader(new FileReader(new File(FileName)));
47  			String line;
48  
49  			while ((line = dataInput.readLine()) != null) {
50  				// buffer.append(cleanLine(line.toLowerCase()));
51  				buffer.append(line);
52  				buffer.append('\n');
53  			}
54  			dataInput.close();
55  		} catch (Exception ex) {
56  			ex.printStackTrace();
57  		}
58  		return buffer.toString();
59  	}

61  	public static void writeFile(String in, String filename) {
62  
63  		try {
64  			File outfile = new File(filename);
65  
66  			if (!(new File(outfile.getParent()).exists())) {
67  				(new File(outfile.getParent())).mkdirs();
68  			}
69  			BufferedWriter bw = new BufferedWriter(new FileWriter(outfile));
70  			bw.write(in);
71  			bw.close();
72  		} catch (Exception e) {
73  			e.printStackTrace();
74  		}
75  	}

77  	public static Double round(double d, int decimalPlace) {
78  
79  		// see the Javadoc about why we use a String in the constructor
80  		// http://java.sun.com/j2se/1.5.0/docs/api/java/math/BigDecimal.html#BigDecimal(double)
81  		BigDecimal bd = new BigDecimal(Double.toString(d));
82  		bd = bd.setScale(decimalPlace, BigDecimal.ROUND_HALF_UP);
83  		return bd.doubleValue();
84  	}

86  	public static Double entropy(double[] p) {
87  
88  		double h = 0;
89  		for (int i = 0; i < p.length; i++) {
90  			h += (p[i] > 0) ? p[i] * Math.log(p[i]) : 0.0;
91  		}
92  		return -h;
93  	}
